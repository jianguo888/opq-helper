# 发送文本消息 群/好友/私聊

URL: http://127.0.0.1:8888/v1/LuaApiCaller

method :POST

# WebAPi请求参数:

params:

{
  'qq':123456, #机器人QQ号
  'funcname':SendMsgV2, #调用方法类型
}

发送好友文本消息 body(json):


{"ToUserUid":123456789,"SendToType":1,"SendMsgType":"TextMsg","Content":"HelloWorld"}

发送群文本消息 body(json):


{"ToUserUid":123456789,"SendToType":2,"SendMsgType":"TextMsg","Content":"HelloWorld"}

发送私聊文本 body(json):


{"ToUserUid":987654321 ,"GroupID":123456789,"SendToType":3,"SendMsgType":"TextMsg","Content":"HelloWorld"}

# ToUserUid 好友QQ/群ID/私聊对象QQ
# GroupID 私聊对象所在的群ID
# SendToType  1 为好友消息 2 发送群消息  3 发送私聊消息

发送群文本特效字体消息 body(json):


{"ToUserUid":123456789,"SendToType":2,"SendMsgType":"TeXiaoTextMsg","Content":"HelloWorld"}

给手机发送消息 body(json):


{"ToUserUid":123456789,"SendToType":2,"SendMsgType":"PhoneMsg","Content":"HelloWorld"}

# LuaApi发送文本

⚠️我就不一一复制发送结构了 WebApi的Json 对应Lua里的table 映射规律是这样的 把Json的字段的引号都去掉 字段值的冒号换成=号 数组[]换成{}⚠️

if data.Content == "SendMsgV2Test" then
        Api.Api_SendMsgV2( --私聊
            CurrentQQ,
            {
                ToUserUid = 987654321,
                GroupID = 987654321,
                SendToType = 3,
                SendMsgType = "TextMsg",
                Content = "HelloWorld"
            }
        )
        Api.Api_SendMsgV2( --好友
            CurrentQQ,
            {
                ToUserUid = 987654321,
                SendToType = 1,
                SendMsgType = "TextMsg",
                Content = "HelloWorld"
            }
        )
        Api.Api_SendMsgV2( --群
            CurrentQQ,
            {
                ToUserUid = 987654321,
                SendToType = 2,
                SendMsgType = "TextMsg",
                Content = "HelloWorld"
            }
        )
    end

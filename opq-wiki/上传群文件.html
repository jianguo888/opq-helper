# 上传群文件

URL: http://127.0.0.1:8888/v1/LuaApiCaller

# 请求信息:

params:

{
 'qq': 123456789,  #bot的QQ
 'funcname': 'SendMsgV2'
}

body(json):

#发送本地文件  Notify 是否通知
{"ToUserUid":123456789,"SendMsgType":"UploadGroupFile","FilePath":"/Users/kar98k/Downloads/0.2.zip","Notify":false}

#发送网络文件  
{"ToUserUid":123456789,"SendMsgType":"UploadGroupFile","FileUrl":"https://cr5.198254.com/winrarx64lhzcx.exe","FileName":"winrar.exe","Notify":false}

#以bas64形式发送文件  
{"ToUserUid":123456789,"SendMsgType":"UploadGroupFile","FileBase64":"==xxsGsNJK=","FileName":"1.mp3","Notify":false}

暂时支持好友 未测试

respone(json)

{"Msg":"已下发上传任务","Ret":0}

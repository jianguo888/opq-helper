# 发送图片消息

URL: http://127.0.0.1:8888/v1/LuaApiCaller

# 请求信息:

params:

{
'qq': 123456789,  #bot的QQ
'funcname': 'SendMsg'
}

body(json):

{
  "toUser":987654321, #发到哪个QQ或者群号
  "sendToType":1,  #自己选择对应会话的数值
  "sendMsgType":"PicMsg",
  "content":"test",   #文字消息
  "groupid":0,  #群号
  "atUser":0,  
  "picUrl":"",  #图片的url,OPQ会自动加Referer
  "picBase64Buf":"",  #base64后的图片
  "fileMd5":"",  #图片MD5,普通图片MD5貌似会过期,表情貌似不会
  "flashPic":true #闪照:仅群聊可用
}

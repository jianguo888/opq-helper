# 上传群图片接口

URL: http://127.0.0.1:8888/v1/GetGroupPicInfo

# 请求信息:

params:

{
 'qq': 123456789,  #bot的QQ
 
}

body(json):

不支持多URL上传 请单次上传
通过URL取图片信息
{
  "PicUrl":"https://photo.16pic.com/00/47/66/16pic_4766855_b.jpg"
}
通过本地取图片信息
{
  "PicPath":"./1.jpg"
}
通过Base64取图片信息
{
  "PicBase64Buf":""
}

response(json)

{
    "ErrMsg":"Ok",
    "PicInfo":{
        "FileId":2992536898,
        "PicMd5":"p6cToURe5t1SHJqGJ83lrw==",
        "PicSize":71982
    },
    "Ret":0
}

# 发送文字消息

URL: http://127.0.0.1:8888/v1/LuaApiCaller

# 请求信息:

params:

{
'qq': 123456789,  #bot的QQ
'funcname': 'SendMsg'
}

body(json):

{
  "toUser":987654321,  #发到哪个QQ或者群号
  "sendToType":1,   #自己选择对应会话的数值
  "sendMsgType":"TextMsg",
  "content":"文字消息",   #要发送的文字内容
  "groupid":0,  #群号
}

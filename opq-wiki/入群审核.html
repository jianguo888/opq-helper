WebApi 获取方式如下

URL: http://127.0.0.1:8888/v1/LuaApiCaller

# 请求信息:

params:

{
'qq': 123456789,  #bot的QQ
'funcname': 'AnswerInviteGroup'
}

body(json):

{
  "Seq":1570980892762275,  #申请进群事件中的Seq
  "Who":123456789,  	 # 申请进群的QQ
  "Flag_7":8192,    	#未知
  "Flag_8":512,    		#未知
  "GroupId":570065685,  # 群号
  "Action":11    # 11:同意 21:拒绝 14:忽略
}

response(json)

{
{"Msg":"已同意该申请","Ret":1003}
}

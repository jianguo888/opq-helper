    # 发送大表情

    URL: http://127.0.0.1:8888/v1/LuaApiCaller

    # 请求信息:

    params:

    {
'qq': 123456789,  #bot的QQ
'funcname': 'SendMsg'
}

    body(json):

    {
  "toUser":987654321,  #发到哪个QQ或者群号
  "sendToType":2,   #发送类型
  "sendMsgType":"ForwordMsg", #消息类型,大表情
  "content":"",   #要发送的文字内容
  "groupid":0,  #0为群消息
  "atUser":0,  #要@的QQ号
  "forwordBuf":"Cghb5Y2W6JCMXRAGGAEiEPT8daec5EJJy8A32ytRrDsogL4MMAM6EDJjNjI3NWIzMmM2M2EwMTlIAFDIAVjIAWoSCgYIrAIQrAIKBgjIARDIAUAB", #表情buf值
  "forwordField":6 #表情类型(6为网络图片)
}

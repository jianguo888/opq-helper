      # Lua 获取方式如下

      function ReceiveFriendMsg(CurrentQQ, data)
    if data.MsgType == "FriendFileMsg" then
        --MsgType FriendFileMsg
        -- Content {"FileID":"36615b65f95b09df4de35ea1c783c368_fde7a4c0-4648-4ec3-b580-2a33006507ac","FileName":"QQ录屏20200601223847.mp4","FileSize":14689605,"Tips":"[好友文件]"}
        file = json.decode(data.Content)
        res =
            Api.Api_CallFunc(
            CurrentQQ,
            "OfflineFilleHandleSvr.pb_ftn_CMD_REQ_APPLY_DOWNLOAD-1200",
            {FileID = file.FileID}
        )
        log.notice(" FileUrl Log\n%s", res.Url)
    end

    return 1
end
